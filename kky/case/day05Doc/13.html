1.命名空间
有时，我们想对事件进行移除。
但对于同名同元素绑定的事件移除往往比较麻烦，
这个时候，可以使用事件的命名空间解决。
点我看控制台! 
			//1.命名空间
			$("#d1").on("click.a",function(){
				console.log("该click事件命名为a");
			});
			$("#d1").on("click.b",function(){
				console.log("该click事件命名为b");
			});
			$("#d1").on("click.c",function(){
				console.log("该click事件命名为c");
			});
			$("#d1").on("click.d",function(){
				console.log("该click事件命名为d");
			});
			$("#d1").off("click.d");
