	4.3 常见命令
		git clone 仓库地址 下载/克隆 这个远程仓库的代码
		git窗口中cd 
		cd 文件夹 进入文件夹
		cd ../ 返回到上一级
	
		git status 查看本地仓库代码是否有变动(添加,修改,删除)
		
		git add .
		git add *
		git add 文件名
		将变动过的文件放入暂存区
		
		git commit -m "注释可以写汉字"  增加/记录 版本
		
		-----------------------------
		将代码推送到gitee
		-----------------------------
		
		git remote -v 查看远程仓库(地址)
		
		git pull origin mater 拉取最新的代码
		
		git push origin master 将代码推送到指定仓库
		origin(一般是origin/也可以改)  
		
		
		分支
		git branch 查看分支
		
		git branch 分支名 创建分支
		
		git checkout 分支名 切换分支
		
		操作数据
		
		git checkout -b 分支名  创建分支,并切换到该分支
		
		git branch -d 分支名 删除分支
		
		-----------------
		git merge 所有合并
		git merge 分支名 指定分支合并 常用 
		
		
		查看提交历史
		
		git log 查看该仓库的git日志,每一个版本
		git log --oneline 日志简写
		git log --reverse --oneline 日志倒序简写
		git log --author=用户 --oneline -5
		
		git blame 文件名  查看文件修改记录
		
		按q退出
		
		git pull origin kky 拉取分支kky 最新的代码
		
		版本回退
		git reset --hard HEAD
		HEAD 表示当前版
		HEAD^ 上一个版本
		HEAD^^ 上上一个版本
		HEAD^^^ 上上上一个版本
		git reset id
		
		HEAD指向的版本就是当前版本，因此，Git允许我们在版本的历史之间穿梭，使用命令git reset --hard commit_id。
		git log
		穿梭前，用git log可以查看提交历史，以便确定要回退到哪个版本。
		
		git reflog
		要重返未来，用git reflog查看命令历史，以便确定要回到未来的哪个版本。
